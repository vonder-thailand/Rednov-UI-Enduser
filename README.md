## Install

```bash
yarn add rednov-ui
```

## Usage

```tsx
import { Typography } from 'rednov-ui'
import 'rednov-ui/dist/index.css'

function App() {
  return (
    <div>
      <Typography.Headline>This is Headline</Typography.Headline>
    </div>
  )
}
```
